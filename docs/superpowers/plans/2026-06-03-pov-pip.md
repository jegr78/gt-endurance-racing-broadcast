# Driver-POV Picture-in-Picture — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an optional ad-hoc driver-POV stream as a small picture-in-picture (384×216, bottom-right) in the OBS Stint scene, pulled by a third independent relay feed (port 53003) and controlled remotely by the Director via a Google-Sheet cell + four Companion buttons.

**Architecture:** Reuse the proven yt-dlp→streamlink pull pipeline as a third, independent `Feed` (NOT part of the A/B ping-pong index), fed by its own Google-Sheet tab `POV` (cell A2). Two new HTTP endpoints (`/pov/reload`, `/pov/stop`) drive it; OBS gets one `ffmpeg_source` (`Feed POV`) in the Stint scene; Companion gets four Page-2 buttons. All changes are additive — existing A/B / schedule / combo behaviour is untouched.

**Tech Stack:** Python 3 stdlib (relay `iro-feeds.py`), OBS scene-collection JSON, Bitfocus Companion config JSON, Google Sheets (gviz CSV), bash/zip packaging.

**Spec:** `docs/superpowers/specs/2026-06-03-pov-pip-design.md`

Create `tools/add_pov_source.py`:

```python
#!/usr/bin/env python3
"""Add the 'Feed POV' PiP source to an OBS scene-collection JSON (idempotent).

It deep-copies the existing 'Feed A' source object and its Stint scene item as
templates, then overrides only the POV-specific fields — so the result always
matches OBS's schema. Re-running is a no-op once 'Feed POV' exists.

Usage: python3 tools/add_pov_source.py <collection.json>
"""
import copy, json, sys

POV_NAME  = "Feed POV"
POV_UUID  = "aaaaaaa3-0000-4000-8000-000000000003"   # A=…001, B=…002, POV=…003
POV_INPUT = "http://127.0.0.1:53003"
POS    = {"x": 1496.0, "y": 644.0}   # bottom-right, above the HUD lower-third
BOUNDS = {"x": 384.0,  "y": 216.0}   # 1/5 size, 16:9


def main(path):
    with open(path, encoding="utf-8") as fh:
        d = json.load(fh)
    srcs = d["sources"]
    if any(s.get("name") == POV_NAME for s in srcs):
        print(f"{path}: '{POV_NAME}' already present — skip"); return
    feedA = next(s for s in srcs if s.get("name") == "Feed A")
    stint = next(s for s in srcs if s.get("id") == "scene" and s.get("name") == "Stint")

    # 1) source object (copy Feed A, override the few POV fields)
    src = copy.deepcopy(feedA)
    src["name"] = POV_NAME
    src["uuid"] = POV_UUID
    src["muted"] = True                                  # default muted (switchable)
    src["settings"] = dict(src.get("settings", {}))
    src["settings"]["input"] = POV_INPUT
    src["settings"]["close_when_inactive"] = True        # no decode while hidden
    src["settings"]["restart_on_activate"] = True        # grab the port on SHOW
    src["settings"]["reconnect_delay_sec"] = 10
    srcs.append(src)

    # 2) Stint scene item (front-most -> append; items are ordered back->front)
    items = stint["settings"]["items"]
    tmpl = next(it for it in items if it.get("name") == "Feed A")
    item = copy.deepcopy(tmpl)
    item["name"] = POV_NAME
    item["source_uuid"] = POV_UUID
    item["visible"] = False
    item["locked"] = False
    item["id"] = max(it.get("id", 0) for it in items) + 1
    item["pos"] = dict(POS)
    item["scale"] = {"x": 1.0, "y": 1.0}
    item["bounds_type"] = 2                               # OBS_BOUNDS_SCALE_INNER = "fit"
    item["bounds"] = dict(BOUNDS)
    items.append(item)

    with open(path, "w", encoding="utf-8") as fh:
        json.dump(d, fh, ensure_ascii=False, indent=4)
    print(f"{path}: added '{POV_NAME}' (hidden, "
          f"{int(BOUNDS['x'])}x{int(BOUNDS['y'])} @ {int(POS['x'])},{int(POS['y'])})")


if __name__ == "__main__":
    if len(sys.argv) != 2:
        sys.exit("usage: add_pov_source.py <collection.json>")
    main(sys.argv[1])
```

Create `tools/strip_companion_pass.py`:

```python
#!/usr/bin/env python3
"""Blank every pass/password field in a Companion config (so the distributed
copy never ships the OBS WebSocket password). Usage: strip_companion_pass.py IN OUT"""
import json, sys


def blank(o):
    if isinstance(o, dict):
        for k, v in o.items():
            if k in ("pass", "password") and isinstance(v, str):
                o[k] = ""
            else:
                blank(v)
    elif isinstance(o, list):
        for x in o:
            blank(x)


def main(src, dst):
    cfg = json.load(open(src, encoding="utf-8"))
    blank(cfg)
    json.dump(cfg, open(dst, "w", encoding="utf-8"), indent=1)
    print(f"stripped {src} -> {dst}")


if __name__ == "__main__":
    if len(sys.argv) != 3:
        sys.exit("usage: strip_companion_pass.py IN OUT")
    main(sys.argv[1], sys.argv[2])
```
